<!DOCTYPE html>

<html>

  <head>

    <meta charset="UTF-8" />

    <title>Redirecting...</title>

    <script>

      const params = new URLSearchParams(window.location.search);

      const type = window.location.pathname.replace("/", ""); // gets "onboarding-return" or "onboarding-refresh"


      const redirectMap = {

        "onboarding-return": "hcarr://onboarding-return",

        "onboarding-refresh": "hcarr://onboarding-refresh"

      };


      const redirectUrl = redirectMap[type];


      if (redirectUrl) {

        window.location.href = redirectUrl;

      } else {

        document.write("<h1>No redirect found.</h1>");

      }

    </script>

  </head>

  <body>

    <h1>Redirecting...</h1>

  </body>

</html>
